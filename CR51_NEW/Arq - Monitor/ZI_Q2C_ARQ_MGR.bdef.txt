managed implementation in class ZBP_I_Q2C_ARQ_MGR unique;
strict ( 2 );

// -----------------------------------------------------------------------
// Root BO: Arquivos de Entrada MGR — Monitor / Reprocessamento
// Tabela : ZTBN_Q2C_ARQ_MGR
// Chave  : (Pedido, Bandeira)
// -----------------------------------------------------------------------
define behavior for ZI_Q2C_ARQ_MGR alias ArqMgr
  persistent table ztbn_q2c_arq_mgr
  lock master
  authorization master ( global )
  etag master Uzeit
{
  // update declarado para permitir MODIFY via EML IN LOCAL MODE nas actions
  // Não exposto na projeção — usuário não edita campos diretamente
  update;

  // Todos os campos são somente leitura para o usuário
  // O preenchimento ocorre exclusivamente via actions ou integração externa
  field ( readonly ) Pedido; Bandeira; TipoDoc; CabecArq; Conteudo;
  field ( readonly ) Status; Tentativas;
  // StatusCriticality é campo calculado no CDS (CASE expression) — não declarar em field(readonly)
  field ( readonly ) UltimoErro; Datum; Uzeit; Ernam;

  action Reprocess result [1] $self;
  action Cancel    result [1] $self;

  // Associação de leitura ao LOG — sem create (LOG é gerenciado pela action handler)
  association _Log { }

  mapping for ztbn_q2c_arq_mgr
  {
    Pedido     = pedido;
    Bandeira   = bandeira;
    TipoDoc    = tipo_doc;
    CabecArq   = cabec_arq;
    Conteudo   = conteudo;
    Status     = status;
    Tentativas = tentativas;
    UltimoErro = ultimo_erro;
    Datum      = datum;
    Uzeit      = uzeit;
    Ernam      = ernam;
  }
}
