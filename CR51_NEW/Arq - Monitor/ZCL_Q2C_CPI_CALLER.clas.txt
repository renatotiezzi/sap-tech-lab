*&---------------------------------------------------------------------*
*& Classe: ZCL_Q2C_CPI_CALLER
*& Responsável por chamar o iFlow CPI para reprocessamento de arquivos.
*&
*& Chamada única a partir de ZBP_I_Q2C_ARQ_MGR (CCIMP, action Reprocess).
*& A assinatura usa a estrutura de tabela plana ztbq2c_arq_mgr para
*& desacoplar do tipo RAP derivado — nenhum caller precisa ser alterado
*& se novos campos forem adicionados ao contrato interno.
*&
*& Para alterar o contrato com o iFlow CPI (payload JSON):
*&   Edite apenas BUILD_PAYLOAD e adicione campos em TY_PAYLOAD_FIELDS.
*&   Não altere CALL_CPI_REPROCESS nem o behavior pool.
*&
*& Destino RFC: CPI_MGR_REPROCESSO (tipo G — HTTP externo, SM59)
*&   Host, porta e path do iFlow já configurados no destino.
*&
*& Payload enviado ao CPI (/http/mgr/reprocessamento):
*&   { "filename": "<tipo_doc>", "conteudo": "<conteudo base64>" }
*&---------------------------------------------------------------------*

CLASS zcl_q2c_cpi_caller DEFINITION
  PUBLIC
  FINAL
  CREATE PUBLIC.

  PUBLIC SECTION.
    " Detalhe do último erro de comunicação CPI.
    " Disponível no CATCH cx_root do behavior pool via cpi_caller->mv_error_detail.
    DATA mv_error_detail TYPE string READ-ONLY.

    " Ponto de entrada único para o behavior pool.
    " is_arq: estrutura plana ztbq2c_arq_mgr (via CORRESPONDING do tipo RAP).
    " ev_status: mensagem de retorno do CPI (opcional — pode ser vazio em sucesso).
    METHODS call_cpi_reprocess
      IMPORTING
        is_arq    TYPE ztbq2c_arq_mgr
      EXPORTING
        ev_status TYPE string
      RAISING
        cx_root.

  PRIVATE SECTION.
    " Destino RFC configurado na SM59 (tipo G - HTTP para sistema externo)
    " Host, porta (443) e path (/http/mgr/reprocessamento) definidos no destino.
    CONSTANTS c_destination TYPE rfcdest VALUE 'CPI_MGR_REPROCESSO'.

    " Campos usados para montar o payload.
    " Para adicionar campo: inclua aqui e use-o em BUILD_PAYLOAD.
    TYPES: BEGIN OF ty_payload_fields,
             bandeira TYPE ztbq2c_arq_mgr-bandeira,
             arquivo  TYPE ztbq2c_arq_mgr-arquivo,
           END OF ty_payload_fields.

    " Único método a alterar quando o contrato do iFlow muda.
    METHODS build_payload
      IMPORTING
        is_fields         TYPE ty_payload_fields
      RETURNING
        VALUE(rv_payload) TYPE string.

    " Executa a chamada HTTP. Erros sinalizados via mv_error_detail.
    " rv_response vazio + mv_error_detail preenchido = chamada com erro.
    METHODS execute_http_call
      IMPORTING
        payload            TYPE string
      RETURNING
        VALUE(rv_response) TYPE string.

ENDCLASS.

CLASS zcl_q2c_cpi_caller IMPLEMENTATION.

  METHOD call_cpi_reprocess.
    CLEAR me->mv_error_detail.

    " 1. Montar payload — dados já disponíveis em is_arq, sem SELECT adicional
    DATA(lv_payload) = build_payload(
      VALUE #(
        bandeira = is_arq-bandeira
        arquivo  = is_arq-arquivo
      )
    ).

    " 2. Enviar ao CPI
    DATA(lv_response) = execute_http_call( lv_payload ).

    IF me->mv_error_detail IS NOT INITIAL.
      RAISE EXCEPTION TYPE cx_dynamic_check.
    ENDIF.

    ev_status = lv_response.

  ENDMETHOD.

  METHOD build_payload.
    " ==========================================================
    " CONTRATO DO IFLOW CPI (/http/mgr/reprocessamento):
    "   "filename" → Bandeira  (ex: FORD, RENAULT)
    "   "conteudo" → Arquivo   (nome do arquivo — ex: NF_FORD_001.xml)
    "
    " Para mudar/adicionar campo:
    "   1. Adicione em ty_payload_fields
    "   2. Preencha no VALUE # em call_cpi_reprocess
    "   3. Inclua na string abaixo
    " ==========================================================
    rv_payload = |\{ "filename": "{ is_fields-bandeira }", | &
                 |"conteudo": "{ is_fields-arquivo }" \}|.
  ENDMETHOD.

  METHOD execute_http_call.

    " --- text-001='Content-Type'  text-002='application/json'
    DATA lv_content_type TYPE string.
    DATA lv_mime_json     TYPE string.
    lv_content_type = text-001.
    lv_mime_json    = text-002.

    DATA: lo_client    TYPE REF TO if_http_client,
          lv_http_code TYPE i,
          lv_reason    TYPE string.

    " 1. Criar cliente HTTP via RFC Destination (SM59: CPI_MGR_REPROCESSO, tipo G)
    cl_http_client=>create_by_destination(
      EXPORTING
        destination              = c_destination
      IMPORTING
        client                   = lo_client
      EXCEPTIONS
        argument_not_found       = 1
        destination_not_found    = 2
        destination_no_authority = 3
        plugin_not_active        = 4
        internal_error           = 5
        OTHERS                   = 6
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Destino RFC '{ c_destination }' não encontrado/acessível (subrc={ sy-subrc })|.
      RETURN.
    ENDIF.

    " 2. Método POST + Content-Type JSON
    lo_client->request->set_method( if_http_request=>co_request_method_post ).
    lo_client->request->set_header_field( name  = lv_content_type
                                          value = lv_mime_json ).

    " 3. Setar body diretamente como string (sem conversão)
*   " Conversão para xstring não necessária — conteudo já está em Base64
*   " lv_body_xstr = cl_bcs_convert=>string_to_xstring( iv_string = payload ).
*   " lo_client->request->set_data( lv_body_xstr ).
    lo_client->request->set_cdata( payload ).

    " 4. Enviar
    lo_client->send(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        OTHERS                     = 3
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Falha de comunicação HTTP ao enviar para CPI (subrc={ sy-subrc })|.
      lo_client->close( ).
      RETURN.
    ENDIF.

    " 5. Receber resposta
    lo_client->receive(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        http_processing_failed     = 3
        OTHERS                     = 4
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Falha ao receber resposta do CPI (subrc={ sy-subrc })|.
      lo_client->close( ).
      RETURN.
    ENDIF.

    " 6. Verificar HTTP status code
    lo_client->response->get_status(
      IMPORTING
        code   = lv_http_code
        reason = lv_reason
    ).

    rv_response = lo_client->response->get_cdata( ).
    lo_client->close( ).

    " HTTP != 2xx = erro de negócio CPI
    IF lv_http_code < 200 OR lv_http_code >= 300.
      me->mv_error_detail = |CPI HTTP { lv_http_code } ({ lv_reason }): { rv_response }|.
      CLEAR rv_response.
    ENDIF.

  ENDMETHOD.

ENDCLASS.
