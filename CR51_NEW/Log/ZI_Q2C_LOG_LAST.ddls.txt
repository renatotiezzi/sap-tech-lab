// -----------------------------------------------------------------------
// Helper interno (sem BDEF) — não expor diretamente em serviço.
// Pressupõe que os registros são inseridos com sy-datum/sy-uzeit em
// ordem cronológica, portanto max(Datum)+max(Uzeit) identifica
// corretamente o registro mais recente por chave.
// Usado exclusivamente por ZI_Q2C_LOG_SUM como base do JOIN.
// -----------------------------------------------------------------------
define view entity ZI_Q2C_LOG_LAST
  as select from ztbq2c_log_mgr
{
  key pedido          as Pedido,
  key bandeira        as Bandeira,
      max( datum )    as LastDatum,
      max( uzeit )    as LastUzeit
}
group by pedido, bandeira
