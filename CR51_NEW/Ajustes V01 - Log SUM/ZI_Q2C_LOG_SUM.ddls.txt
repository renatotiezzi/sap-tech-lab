// -----------------------------------------------------------------------
// Entidade raiz do App 2 (LOG Viewer).
// Chave: (Pedido, Bandeira) — uma linha por combinação de chave.
// Exibe os campos da ÚLTIMA execução registrada (UltDatum, UltEtapa...).
//
// Usa ZI_Q2C_LOG_LAST para identificar o último registro (max datum+uzeit),
// depois faz INNER JOIN para puxar todos os campos do registro.
// _Detail → ZI_Q2C_LOG_DET: todas as execuções para aquela chave.
// Composição pai→filho: filho não é acessível como root entity independente.
// -----------------------------------------------------------------------
define root view entity ZI_Q2C_LOG_SUM
  as select from ZI_Q2C_LOG_LAST as last
  inner join    ztbq2c_log_mgr   as log
    on  last.Pedido    = log.pedido
    and last.Bandeira  = log.bandeira
    and last.LastDatum = log.datum
    and last.LastUzeit = log.uzeit

  composition [0..*] of ZI_Q2C_LOG_DET as _Detail
{
  key log.pedido   as Pedido,
  key log.bandeira as Bandeira,
      log.datum    as UltDatum,
      log.uzeit    as UltUzeit,
      log.etapa    as UltEtapa,
      log.mensagem as UltMensagem,
      log.ernam    as UltErnam,

      /* Association */
      _Detail
}
