@AbapCatalog.viewEnhancementCategory: [#NONE]
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Arquivos de Entrada MGR - Interface View LR'
@Metadata.ignorePropagatedAnnotations: true
@ObjectModel.usageType: {
  serviceQuality: #X,
  sizeCategory: #S,
  dataClass: #MIXED
}
define root view entity ZI_Q2C_ARQLR_MGR
  as select from ztbq2c_arq_mgr as arq
  left outer join ztbq2c_log_mgr as log
    on  log.pedido   = arq.pedido
    and log.bandeira = arq.bandeira
{
  key arq.pedido    as Pedido,
  key arq.bandeira  as Bandeira,
      arq.tipo_doc  as TipoDoc,
      arq.cabec_arq as CabecArq,
      arq.conteudo  as Conteudo,
      arq.status    as Status,
      arq.tentativas as Tentativas,
      arq.datum     as Datum,
      arq.uzeit     as Uzeit,
      arq.ernam     as Ernam,
      log.id_ref    as LogIdRef,
      log.etapa     as LogEtapa,
      log.mensagem  as LogMensagem,
      log.datum     as LogDatum,
      log.uzeit     as LogUzeit,
      log.ernam     as LogErnam
}
