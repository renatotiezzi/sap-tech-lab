managed implementation in class zbp_i_q2c_arq_mgr unique;
strict ( 2 );

// -----------------------------------------------------------------------
// Root BO: Arquivos de Entrada MGR
// -----------------------------------------------------------------------
define behavior for ZI_Q2C_ARQ_MGR alias ArqMgr
persistent table ztbq2c_arq_mgr
lock master
authorization master ( global )
etag master Uzeit
{
  create;
  update;
  delete;

  // Admin fields filled by determination - not editable by user
  field ( readonly ) Datum, Uzeit, Ernam;
  field ( readonly : update ) Pedido, Bandeira;

  // Determinação disparada apenas em campos de negócio.
  // NÃO incluir Datum/Uzeit/Ernam aqui — são os campos que a determinação
  // grava. Incluí-los causaria loop infinito (LCX_ABAP_BEHV_DETVAL_ERROR).
  determination set_admin_fields on modify
    { field Pedido; field Bandeira; field TipoDoc; field Status;
      field Conteudo; field CabecArq; field Tentativas; }

  // Actions defined here - exposed via projection BDEF
  action Reprocess result [1] $self;
  action Cancel    result [1] $self;

  // Composition to Log child entity
  // Nota: update do LOG é feito diretamente na entidade LogMgr (não via associação do pai)
  association _Log { create; }

  mapping for ztbq2c_arq_mgr
  {
    Pedido     = pedido;
    Bandeira   = bandeira;
    TipoDoc    = tipo_doc;
    CabecArq   = cabec_arq;
    Conteudo   = conteudo;
    Status     = status;
    Tentativas = tentativas;
    Datum      = datum;
    Uzeit      = uzeit;
    Ernam      = ernam;
  }
}

// -----------------------------------------------------------------------
// Child BO: Log de Processamento MGR
// -----------------------------------------------------------------------
define behavior for ZI_Q2C_LOG_MGR alias LogMgr
persistent table ztbq2c_log_mgr
lock dependent by _Arq
authorization dependent by _Arq
etag master Uzeit
{
  create;
  update;
  delete;

  // Admin fields filled by determination - not editable by user
  field ( readonly ) Datum, Uzeit, Ernam;

  // Mesma regra: disparar apenas em campos de negócio do LOG.
  determination set_admin_fields on modify
    { field Pedido; field Bandeira; field IdRef; field Etapa; field Mensagem; }

  association _Arq;

  mapping for ztbq2c_log_mgr
  {
    Pedido   = pedido;
    Bandeira = bandeira;
    IdRef    = id_ref;
    Etapa    = etapa;
    Mensagem = mensagem;
    Datum    = datum;
    Uzeit    = uzeit;
    Ernam    = ernam;
  }
}
