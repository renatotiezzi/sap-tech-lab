*&---------------------------------------------------------------------*
*& Class ZCL_Q2C_CPI_CALLER
*&---------------------------------------------------------------------*
*& Responsável por chamar o CPI para reprocessamento de arquivos
*& - Prepara payload JSON
*& - Chamada HTTP/API
*& - Retorna status
*&---------------------------------------------------------------------*

CLASS zcl_q2c_cpi_caller DEFINITION
  PUBLIC
  FINAL
  CREATE PUBLIC.

  PUBLIC SECTION.
    METHODS call_cpi_reprocess
      IMPORTING
        pedido    TYPE ztbq2c_arq_mgr-pedido
        bandeira  TYPE ztbq2c_arq_mgr-bandeira
        conteudo  TYPE ztbq2c_arq_mgr-conteudo
        tipo_doc  TYPE ztbq2c_arq_mgr-tipo_doc
      EXPORTING
        ev_status TYPE string.

  PRIVATE SECTION.
    " Destino RFC configurado na SM59 (tipo G - HTTP p/ sistema externo)
    " Host/Path já definidos no destino; não codificar URL aqui.
    CONSTANTS c_destination TYPE string VALUE 'CPI_MGR_REPROCESSO'.

    METHODS prepare_payload
      IMPORTING
        conteudo  TYPE ztbq2c_arq_mgr-conteudo
        tipo_doc  TYPE ztbq2c_arq_mgr-tipo_doc
      RETURNING
        VALUE(rs_payload) TYPE string.

    METHODS execute_http_call
      IMPORTING
        payload   TYPE string
      RETURNING
        VALUE(rs_response) TYPE string.

ENDCLASS.

CLASS zcl_q2c_cpi_caller IMPLEMENTATION.

  METHOD call_cpi_reprocess.
    DATA payload  TYPE string.
    DATA response TYPE string.

    TRY.
        " 1. Montar payload JSON para o iFlow de reprocessamento
        payload = prepare_payload(
          conteudo = conteudo
          tipo_doc = tipo_doc
        ).

        " 2. Executar chamada HTTP via RFC Destination CPI_MGR_REPROCESSO
        response = execute_http_call( payload ).

        " 3. Retornar body da resposta como status de sucesso
        ev_status = response.

    CATCH cx_root INTO DATA(exc).
        " Propaga mensagem de erro para o behavior pool registrar no LOG
        ev_status = |Erro CPI: { exc->get_text( ) }|.
        RAISE EXCEPTION exc.
    ENDTRY.

  ENDMETHOD.

  METHOD prepare_payload.
    " Payload esperado pelo iFlow CPI /http/mgr/reprocessamento:
    " {
    "   "filename": "<tipo_doc>",     <- código do arquivo (ex: PMDREN, PMDFORD)
    "   "conteudo": "<base64>"        <- linhas do pedido em Base64 (campo ZTBQ2C_ARQ_MGR-conteudo)
    " }
    rs_payload = |\{ "filename": "{ tipo_doc }", "conteudo": "{ conteudo }" \}|.

  ENDMETHOD.

  METHOD execute_http_call.
    DATA: lo_client    TYPE REF TO if_http_client,
          lv_body_xstr TYPE xstring,
          lv_http_code TYPE i,
          lv_reason    TYPE string,
          lv_response  TYPE string.

    " 1. Criar cliente HTTP via RFC Destination (SM59: CPI_MGR_REPROCESSO, tipo G)
    "    Host e path já configurados no destino — não codificar URL aqui.
    cl_http_client=>create_by_destination(
      EXPORTING
        destination              = c_destination
      IMPORTING
        client                   = lo_client
      EXCEPTIONS
        argument_not_found       = 1
        destination_not_found    = 2
        destination_no_authority = 3
        plugin_not_active        = 4
        internal_error           = 5
        OTHERS                   = 6
    ).
    IF sy-subrc <> 0.
      RAISE EXCEPTION NEW cx_abap_illegal_argument(
        parameter = |Destino RFC '{ c_destination }' não encontrado/acessível (sy-subrc={ sy-subrc })|
      ).
    ENDIF.

    " 2. Método POST
    lo_client->request->set_method( if_http_request=>co_request_method_post ).

    " 3. Content-Type JSON
    lo_client->request->set_header_field(
      name  = 'Content-Type'
      value = 'application/json'
    ).

    " 4. Converter payload string → xstring e setar como body
    cl_bcs_convert=>string_to_xstring(
      EXPORTING iv_string  = payload
      RECEIVING ev_xstring = lv_body_xstr
    ).
    lo_client->request->set_data( lv_body_xstr ).

    " 5. Enviar requisição
    lo_client->send(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        OTHERS                     = 3
    ).
    IF sy-subrc <> 0.
      lo_client->close( ).
      RAISE EXCEPTION NEW cx_abap_illegal_argument(
        parameter = |Falha de comunicação HTTP ao enviar para CPI (sy-subrc={ sy-subrc })|
      ).
    ENDIF.

    " 6. Aguardar e receber resposta
    lo_client->receive(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        http_processing_failed     = 3
        OTHERS                     = 4
    ).
    IF sy-subrc <> 0.
      lo_client->close( ).
      RAISE EXCEPTION NEW cx_abap_illegal_argument(
        parameter = |Falha ao receber resposta do CPI (sy-subrc={ sy-subrc })|
      ).
    ENDIF.

    " 7. Verificar HTTP status code
    lo_client->response->get_status(
      IMPORTING
        code   = lv_http_code
        reason = lv_reason
    ).

    " 8. Ler body da resposta
    lv_response = lo_client->response->get_cdata( ).

    " 9. Fechar conexão
    lo_client->close( ).

    " 10. 2xx = sucesso; outros = erro com detalhes para o LOG
    IF lv_http_code >= 200 AND lv_http_code < 300.
      rs_response = COND #(
        WHEN lv_response IS NOT INITIAL
        THEN lv_response
        ELSE |CPI OK (HTTP { lv_http_code })|
      ).
    ELSE.
      RAISE EXCEPTION NEW cx_abap_illegal_argument(
        parameter = |CPI retornou HTTP { lv_http_code } ({ lv_reason }): { lv_response }|
      ).
    ENDIF.

  ENDMETHOD.

ENDCLASS.
