managed;

define behavior for ZI_Q2C_LOG_MGR alias LogMgr
  persistent table ztbq2c_log_mgr
  lock master
{
  // create: permite que o CPI insira linhas via inbound OData (ZC_Q2C_LOG_INB)
  // A projeção UI (ZC_Q2C_LOG_MGR_APP) não expõe create — somente leitura para o usuário
  create;

  mapping for ztbq2c_log_mgr
  {
    Pedido   = pedido;
    Bandeira = bandeira;
    IdRef    = id_ref;
    Datum    = datum;
    Uzeit    = uzeit;
    Etapa    = etapa;
    Mensagem = mensagem;
    Ernam    = ernam;
  }
}
