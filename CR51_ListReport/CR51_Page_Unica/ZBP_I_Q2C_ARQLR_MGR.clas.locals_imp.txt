*"* local helper classes, interface definitions and type definitions

CLASS lhc_arq_mgr DEFINITION INHERITING FROM cl_abap_behavior_handler.
  PRIVATE SECTION.
    METHODS set_admin_fields FOR DETERMINE ON MODIFY
      IMPORTING keys FOR arqmgr~set_admin_fields.

    METHODS reprocess FOR MODIFY
      IMPORTING keys FOR ACTION arqmgr~reprocess
      RESULT result.

    METHODS cancel FOR MODIFY
      IMPORTING keys FOR ACTION arqmgr~cancel
      RESULT result.

    METHODS upsert_log
      IMPORTING
        pedido   TYPE ztbq2c_arq_mgr-pedido
        bandeira TYPE ztbq2c_arq_mgr-bandeira
        etapa    TYPE ztbq2c_log_mgr-etapa
        mensagem TYPE ztbq2c_log_mgr-mensagem
        id_ref   TYPE ztbq2c_log_mgr-id_ref OPTIONAL.
ENDCLASS.

CLASS lhc_arq_mgr IMPLEMENTATION.

  METHOD set_admin_fields.
    READ ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
      ENTITY arqmgr
        FIELDS ( Pedido Bandeira )
        WITH CORRESPONDING #( keys )
      RESULT DATA(entries).

    DATA updates TYPE TABLE FOR UPDATE zi_q2c_arqlr_mgr\\arqmgr.

    LOOP AT entries INTO DATA(entry).
      APPEND VALUE #(
        %tky     = entry-%tky
        datum    = sy-datum
        uzeit    = sy-uzeit
        ernam    = sy-uname
        %control = VALUE #(
          datum = if_abap_behv=>mk-on
          uzeit = if_abap_behv=>mk-on
          ernam = if_abap_behv=>mk-on
        )
      ) TO updates.
    ENDLOOP.

    MODIFY ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
      ENTITY arqmgr
        UPDATE FIELDS ( Datum Uzeit Ernam )
        WITH updates.
  ENDMETHOD.

  METHOD reprocess.
    READ ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(entities).

    DATA(cpi_caller) = NEW zcl_q2c_cpi_caller( ).

    LOOP AT entities INTO DATA(entity).
      TRY.
          cpi_caller->call_cpi_reprocess(
            EXPORTING
              pedido   = entity-Pedido
              bandeira = entity-Bandeira
              conteudo = entity-Conteudo
              tipo_doc = entity-TipoDoc
            IMPORTING
              ev_status = DATA(cpi_status)
          ).

          MODIFY ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
            ENTITY arqmgr
              UPDATE FIELDS ( Tentativas Status )
              WITH VALUE #( (
                %key       = entity-%key
                Tentativas = entity-Tentativas + 1
                Status     = 'PROCESSADO'
              ) ).

          upsert_log(
            pedido   = entity-Pedido
            bandeira = entity-Bandeira
            etapa    = 'REPROCESSAMENTO'
            mensagem = COND #( WHEN cpi_status IS NOT INITIAL
                               THEN cpi_status
                               ELSE 'Reprocessado com sucesso' )
          ).
        CATCH cx_root INTO DATA(exc).
          MODIFY ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
            ENTITY arqmgr
              UPDATE FIELDS ( Tentativas Status )
              WITH VALUE #( (
                %key       = entity-%key
                Tentativas = entity-Tentativas + 1
                Status     = 'ERRO'
              ) ).

          upsert_log(
            pedido   = entity-Pedido
            bandeira = entity-Bandeira
            etapa    = 'REPROCESSAMENTO'
            mensagem = |Erro: { exc->get_text( ) }|
          ).
      ENDTRY.
    ENDLOOP.

    READ ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(result_entities).

    result = VALUE #( FOR entry IN result_entities (
      %key   = entry-%key
      %param = entry
    ) ).
  ENDMETHOD.

  METHOD cancel.
    READ ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(entities).

    MODIFY ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
      ENTITY arqmgr
        UPDATE FIELDS ( Status )
        WITH VALUE #( FOR key IN keys (
          %key   = key-%key
          Status = 'CANCELADO'
        ) ).

    LOOP AT entities INTO DATA(entity).
      upsert_log(
        pedido   = entity-Pedido
        bandeira = entity-Bandeira
        etapa    = 'CANCELAMENTO'
        mensagem = 'Registro cancelado manualmente'
      ).
    ENDLOOP.

    READ ENTITIES OF zi_q2c_arqlr_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(result_entities).

    result = VALUE #( FOR entry IN result_entities (
      %key   = entry-%key
      %param = entry
    ) ).
  ENDMETHOD.

  METHOD upsert_log.
    UPDATE ztbq2c_log_mgr
      SET id_ref    = @id_ref,
          etapa     = @etapa,
          mensagem  = @mensagem,
          datum     = @sy-datum,
          uzeit     = @sy-uzeit,
          ernam     = @sy-uname
      WHERE pedido   = @pedido
        AND bandeira = @bandeira.

    IF sy-subrc <> 0.
      INSERT ztbq2c_log_mgr FROM @( VALUE #(
        pedido   = pedido
        bandeira = bandeira
        id_ref   = id_ref
        etapa    = etapa
        mensagem = mensagem
        datum    = sy-datum
        uzeit    = sy-uzeit
        ernam    = sy-uname
      ) ).
    ENDIF.
  ENDMETHOD.

ENDCLASS.
