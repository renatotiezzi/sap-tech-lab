*&---------------------------------------------------------------------*
*& Class ZCL_Q2C_CPI_CALLER
*&---------------------------------------------------------------------*
*& Responsável por chamar o CPI para reprocessamento de arquivos
*& - Prepara payload JSON
*& - Chamada HTTP/API
*& - Retorna status
*&---------------------------------------------------------------------*

CLASS zcl_q2c_cpi_caller DEFINITION
  PUBLIC
  FINAL
  CREATE PUBLIC.

  PUBLIC SECTION.
    " Detalhe do último erro de comunicação CPI.
    " Disponível no CATCH cx_root do behavior pool via cpi_caller->mv_error_detail.
    DATA mv_error_detail TYPE string READ-ONLY.

    METHODS call_cpi_reprocess
      IMPORTING
        pedido    TYPE ztbq2c_arq_mgr-pedido
        bandeira  TYPE ztbq2c_arq_mgr-bandeira
        conteudo  TYPE ztbq2c_arq_mgr-conteudo
        tipo_doc  TYPE ztbq2c_arq_mgr-tipo_doc
      EXPORTING
        ev_status TYPE string.

  PRIVATE SECTION.
    " Destino RFC configurado na SM59 (tipo G - HTTP p/ sistema externo)
    " Host/Path já definidos no destino; não codificar URL aqui.
    " rfcdest = CHAR 32, tipo nativo do parâmetro destination do cl_http_client
    CONSTANTS c_destination TYPE rfcdest VALUE 'CPI_MGR_REPROCESSO'.

    METHODS prepare_payload
      IMPORTING
        conteudo  TYPE ztbq2c_arq_mgr-conteudo
        tipo_doc  TYPE ztbq2c_arq_mgr-tipo_doc
      RETURNING
        VALUE(rs_payload) TYPE string.

    " execute_http_call não lança exceções — erros sinalizados via mv_error_detail.
    " rs_response vazio + mv_error_detail preenchido = chamada com erro.
    METHODS execute_http_call
      IMPORTING
        payload   TYPE string
      RETURNING
        VALUE(rs_response) TYPE string.

ENDCLASS.

CLASS zcl_q2c_cpi_caller IMPLEMENTATION.

  METHOD call_cpi_reprocess.
    " Resetar detalhe de erro da chamada anterior
    CLEAR me->mv_error_detail.

    " 1. Montar payload JSON
    DATA(lv_payload) = prepare_payload(
      conteudo = conteudo
      tipo_doc = tipo_doc
    ).

    " 2. Executar chamada HTTP — erros sinalizados via mv_error_detail
    DATA(lv_response) = execute_http_call( lv_payload ).

    " 3. Se mv_error_detail preenchido: execute_http_call detectou erro
    "    cx_sy_no_handler é no-check (não precisa de RAISING) e é
    "    capturado pelo CATCH cx_root do behavior pool.
    IF me->mv_error_detail IS NOT INITIAL.
      RAISE EXCEPTION TYPE cx_sy_no_handler.
    ENDIF.

    " 4. Sucesso: retornar body da resposta CPI
    ev_status = lv_response.

  ENDMETHOD.

  METHOD prepare_payload.
    " Payload esperado pelo iFlow CPI /http/mgr/reprocessamento:
    " {
    "   "filename": "<tipo_doc>",     <- código do arquivo (ex: PMDREN, PMDFORD)
    "   "conteudo": "<base64>"        <- conteudo do pedido codificado em Base64
    " }

    " conteudo está na tabela como string (linhas do arquivo) — precisa ser Base64
    DATA lv_xstr TYPE xstring.
    DATA lv_conteudo_b64 TYPE string.

    cl_bcs_convert=>string_to_xstring(
      EXPORTING iv_string  = conteudo
      RECEIVING ev_xstring = lv_xstr
    ).

    " SSFC_BASE64_ENCODE: xstring → Base64 string (disponível em todas as versões SAP)
    CALL FUNCTION 'SSFC_BASE64_ENCODE'
      EXPORTING
        bindata       = lv_xstr
      IMPORTING
        b64data       = lv_conteudo_b64
      EXCEPTIONS
        ssf_krn_error = 1
        OTHERS        = 2.
    IF sy-subrc <> 0.
      lv_conteudo_b64 = conteudo. " fallback: envia sem codificar
    ENDIF.

    rs_payload = |\{ "filename": "{ tipo_doc }", "conteudo": "{ lv_conteudo_b64 }" \}|.

  ENDMETHOD.

  METHOD execute_http_call.
    DATA: lo_client    TYPE REF TO if_http_client,
          lv_body_xstr TYPE xstring,
          lv_http_code TYPE i,
          lv_reason    TYPE string.

    " 1. Criar cliente HTTP via RFC Destination (SM59: CPI_MGR_REPROCESSO, tipo G)
    "    Host e path (Port 443 + /http/mgr/reprocessamento) já definidos no destino.
    cl_http_client=>create_by_destination(
      EXPORTING
        destination              = c_destination
      IMPORTING
        client                   = lo_client
      EXCEPTIONS
        argument_not_found       = 1
        destination_not_found    = 2
        destination_no_authority = 3
        plugin_not_active        = 4
        internal_error           = 5
        OTHERS                   = 6
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Destino RFC '{ c_destination }' não encontrado/acessível (subrc={ sy-subrc })|.
      RETURN.
    ENDIF.

    " 2. Método POST
    lo_client->request->set_method( if_http_request=>co_request_method_post ).

    " 3. Content-Type JSON
    lo_client->request->set_header_field(
      name  = 'Content-Type'
      value = 'application/json'
    ).

    " 4. Converter payload JSON (string) → xstring e setar como body
    " O body é JSON puro; apenas o campo conteudo dentro do JSON é Base64
    cl_bcs_convert=>string_to_xstring(
      EXPORTING iv_string  = payload
      RECEIVING ev_xstring = lv_body_xstr
    ).
    lo_client->request->set_data( lv_body_xstr ).

    " 5. Enviar requisição
    lo_client->send(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        OTHERS                     = 3
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Falha de comunicação HTTP ao enviar para CPI (subrc={ sy-subrc })|.
      lo_client->close( ).
      RETURN.
    ENDIF.

    " 6. Receber resposta
    lo_client->receive(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        http_processing_failed     = 3
        OTHERS                     = 4
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Falha ao receber resposta do CPI (subrc={ sy-subrc })|.
      lo_client->close( ).
      RETURN.
    ENDIF.

    " 7. Verificar HTTP status code
    lo_client->response->get_status(
      IMPORTING
        code   = lv_http_code
        reason = lv_reason
    ).

    " 8. Ler body e fechar conexão
    rs_response = lo_client->response->get_cdata( ).
    lo_client->close( ).

    " 9. HTTP != 2xx = erro de negócio CPI; sinalizar via mv_error_detail
    IF lv_http_code < 200 OR lv_http_code >= 300.
      me->mv_error_detail = |CPI HTTP { lv_http_code } ({ lv_reason }): { rs_response }|.
      CLEAR rs_response.
    ENDIF.

  ENDMETHOD.

ENDCLASS.
