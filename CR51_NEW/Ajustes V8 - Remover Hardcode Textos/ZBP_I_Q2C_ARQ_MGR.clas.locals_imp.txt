*----------------------------------------------------------------------*
* Local Handler: ArqMgr
* Behavior implementation para ZI_Q2C_ARQ_MGR
*
* Arquitetura:
*  - ARQ e LOG são BOs completamente independentes (sem composition)
*  - Cada reprocessamento gera INSERT de nova linha em ZTBQ2C_LOG_MGR
*  - ARQ possui campo ULTIMO_ERRO (STRING) — exibido no cockpit (atualizado a cada erro)
*  - INSERT no LOG é feito via ABAP SQL direto (LOG é read-only no RAP)
*
* V4 - Erro CPI:
*  - Chamada CPI movida para ZFM_Q2C_CPI_CALL_ASYNC (RFC-enabled)
*  - Handler dispara FM via STARTING NEW TASK → retorna imediatamente
*  - Lock RAP é liberado ANTES da chamada HTTP ao CPI
*  - Resolve HTTP 423 causado pelo lock ativo durante callback do CPI
*
* V7 - Fix Async Reprocess:
*  - V6 havia removido o STARTING NEW TASK e voltado chamada síncrona (regressão)
*  - Restaurado padrão assíncrono do V4 (CALL FUNCTION STARTING NEW TASK)
*  - Adicionado bloqueio de status PROCESSADO (objetivo original do V6)
*  - Bloqueios: CANCELADO e PROCESSADO → reprocessamento não permitido
*
* V8 - Remover hardcode de textos:
*  - Literais de texto declarados como CONSTANTS na PRIVATE SECTION da classe
*----------------------------------------------------------------------*
CLASS lhc_arq_mgr DEFINITION INHERITING FROM cl_abap_behavior_handler.
  PRIVATE SECTION.

    CONSTANTS:
      c_status_cancelado   TYPE c LENGTH 20  VALUE 'CANCELADO',
      c_status_processado  TYPE c LENGTH 20  VALUE 'PROCESSADO',
      c_status_em_proc     TYPE c LENGTH 20  VALUE 'EM_PROCESSAMENTO',
      c_etapa_reprocess    TYPE c LENGTH 20  VALUE 'REPROCESSAMENTO',
      c_etapa_cancel       TYPE c LENGTH 20  VALUE 'CANCELAMENTO',
      c_msg_reprocess_init TYPE string       VALUE 'Reprocessamento iniciado',
      c_msg_cancel_manual  TYPE string       VALUE 'Registro cancelado manualmente',
      c_task_cpi_reprocess TYPE c LENGTH 20  VALUE 'CPI_REPROCESS'.

    METHODS reprocess FOR MODIFY
      IMPORTING keys FOR ACTION arqmgr~reprocess
      RESULT result.

    METHODS cancel FOR MODIFY
      IMPORTING keys FOR ACTION arqmgr~cancel
      RESULT result.

    METHODS get_global_authorizations FOR GLOBAL AUTHORIZATION
      IMPORTING REQUEST requested_authorizations FOR ARQMGR RESULT result.

    " Insere nova linha no LOG (sempre INSERT — nunca UPDATE/UPSERT)
    " Garante histórico completo de todas as tentativas
    METHODS insert_log
      IMPORTING
        pedido   TYPE ztbq2c_arq_mgr-pedido
        bandeira TYPE ztbq2c_arq_mgr-bandeira
        etapa    TYPE ztbq2c_log_mgr-etapa
        mensagem TYPE ztbq2c_log_mgr-mensagem.

ENDCLASS.

CLASS lhc_arq_mgr IMPLEMENTATION.

  METHOD get_global_authorizations.
    result = VALUE #(
      %update              = if_abap_behv=>auth-allowed
      %action-Reprocess    = if_abap_behv=>auth-allowed
      %action-Cancel       = if_abap_behv=>auth-allowed
    ).
  ENDMETHOD.

  METHOD reprocess.

    " 1. Ler dados completos das entidades selecionadas
    READ ENTITIES OF zi_q2c_arq_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(entities)
      FAILED DATA(read_failed).

    LOOP AT entities INTO DATA(entity).

      " 2a. Validar: não reprocessar CANCELADO
      IF entity-Status = c_status_cancelado.
        APPEND VALUE #( %key = entity-%key ) TO failed-arqmgr.
        APPEND VALUE #(
          %key = entity-%key
          %msg = new_message_with_text(
            severity = if_abap_behv_message=>severity-error
            text     = |Registro { entity-Pedido }/{ entity-Bandeira } não pode ser reprocessado (status: { entity-Status })|
          )
        ) TO reported-arqmgr.
        CONTINUE.
      ENDIF.

      " 2b. Validar: não reprocessar PROCESSADO
      IF entity-Status = c_status_processado.
        APPEND VALUE #( %key = entity-%key ) TO failed-arqmgr.
        APPEND VALUE #(
          %key = entity-%key
          %msg = new_message_with_text(
            severity = if_abap_behv_message=>severity-error
            text     = |Registro { entity-Pedido }/{ entity-Bandeira } já foi processado — reprocessamento não permitido|
          )
        ) TO reported-arqmgr.
        CONTINUE.
      ENDIF.

      " 3. Incrementar tentativas + marcar EM_PROCESSAMENTO
      "    Status atualizado aqui para feedback imediato ao usuário no Fiori.
      "    ZFM_Q2C_CPI_CALL_ASYNC atualizará para PROCESSADO ou ERRO ao concluir.
      MODIFY ENTITIES OF zi_q2c_arq_mgr IN LOCAL MODE
        ENTITY arqmgr
          UPDATE FIELDS ( Tentativas Status Datum Uzeit Ernam )
          WITH VALUE #( (
            %key       = entity-%key
            Tentativas = entity-Tentativas + 1
            Status     = c_status_em_proc
            Datum      = sy-datum
            Uzeit      = sy-uzeit
            Ernam      = sy-uname
          ) )
        FAILED DATA(upd_failed)
        REPORTED DATA(upd_reported).

      " 4. LOG: registrar início do reprocessamento
      insert_log(
        pedido   = entity-Pedido
        bandeira = entity-Bandeira
        etapa    = c_etapa_reprocess
        mensagem = c_msg_reprocess_init
      ).

      " 5. Disparar chamada CPI em nova work process (assíncrono).
      "    STARTING NEW TASK retorna IMEDIATAMENTE — o lock RAP é liberado
      "    antes de qualquer chamada HTTP ao CPI, eliminando o HTTP 423.
      "    ZFM_Q2C_CPI_CALL_ASYNC é responsável por:
      "      - Chamar ZCL_Q2C_CPI_CALLER->call_cpi_reprocess
      "      - Atualizar Status / UltimoErro em ZTBQ2C_ARQ_MGR via SQL direto
      "      - Inserir linha de resultado em ZTBQ2C_LOG_MGR
      "      - COMMIT WORK
      CALL FUNCTION 'ZFM_Q2C_CPI_CALL_ASYNC'
        STARTING NEW TASK c_task_cpi_reprocess
        EXPORTING
          iv_pedido   = entity-Pedido
          iv_bandeira = entity-Bandeira.

    ENDLOOP.

    " 6. Ler estado final e retornar (obrigatório para result [1] $self)
    "    Neste ponto o status já está EM_PROCESSAMENTO — o FM async
    "    atualizará para PROCESSADO/ERRO em segundo plano.
    READ ENTITIES OF zi_q2c_arq_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(result_entities).

    result = VALUE #( FOR res IN result_entities (
      %key   = res-%key
      %param = res
    ) ).

  ENDMETHOD.

  METHOD cancel.

    " 1. Ler dados atuais
    READ ENTITIES OF zi_q2c_arq_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(entities).

    LOOP AT entities INTO DATA(entity).

      " 2. Validar: não cancelar PROCESSADO
      IF entity-Status = c_status_processado.
        APPEND VALUE #( %key = entity-%key ) TO failed-arqmgr.
        APPEND VALUE #(
          %key = entity-%key
          %msg = new_message_with_text(
            severity = if_abap_behv_message=>severity-error
            text     = |Registro { entity-Pedido }/{ entity-Bandeira } já processado — cancelamento não permitido|
          )
        ) TO reported-arqmgr.
        CONTINUE.
      ENDIF.

      " 3. STATUS = CANCELADO
      MODIFY ENTITIES OF zi_q2c_arq_mgr IN LOCAL MODE
        ENTITY arqmgr
          UPDATE FIELDS ( Status Datum Uzeit Ernam )
          WITH VALUE #( (
            %key  = entity-%key
            Status = c_status_cancelado
            Datum  = sy-datum
            Uzeit  = sy-uzeit
            Ernam  = sy-uname
          ) )
        FAILED DATA(upd_failed)
        REPORTED DATA(upd_reported).

      " 4. LOG: registrar cancelamento
      insert_log(
        pedido   = entity-Pedido
        bandeira = entity-Bandeira
        etapa    = c_etapa_cancel
        mensagem = c_msg_cancel_manual
      ).

    ENDLOOP.

    " 5. Ler estado final e retornar
    READ ENTITIES OF zi_q2c_arq_mgr IN LOCAL MODE
      ENTITY arqmgr
        ALL FIELDS
        WITH CORRESPONDING #( keys )
      RESULT DATA(result_entities).

    result = VALUE #( FOR res IN result_entities (
      %key   = res-%key
      %param = res
    ) ).

  ENDMETHOD.

  METHOD insert_log.
    " Insere SEMPRE nova linha — nunca sobrescreve.
    " Garante histórico completo de todas as tentativas por Pedido + Bandeira.
    " INSERT via ABAP SQL direto pois LOG é BO independente (read-only no RAP).

    INSERT INTO ztbq2c_log_mgr VALUES @(
      VALUE ztbq2c_log_mgr(
        pedido   = pedido
        bandeira = bandeira
        id_ref   = CONV #( |{ sy-datum+4(4) }{ sy-uzeit }| )
        datum    = sy-datum
        uzeit    = sy-uzeit
        etapa    = etapa
        mensagem = mensagem
        ernam    = sy-uname
      )
    ).

    " id_ref é gerado como MMDDHHMMSS (10 chars) — único dentro do mesmo segundo.
    " Se dois reprocessamentos ocorrerem no mesmo segundo, o segundo INSERT falha.

    ENDIF.

  ENDMETHOD.

ENDCLASS.
