@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Interface View - Log MGR Q2C'

define view entity ZI_Q2C_LOG
  as select from ztbq2c_log_mgr

  association to parent ZI_Q2C_ARQ as _Arq
    on $projection.IdRef = _Arq.IdArq

{
  key id_log     as IdLog,

      id_ref     as IdRef,
      etapa      as Etapa,
      mensagem   as Mensagem,

      datum      as Datum,
      uzeit      as Uzeit,
      ernam      as Ernam,

      /* Texto da etapa */
      case etapa
        when '1' then 'Leitura'
        when '2' then 'Validação'
        when '3' then 'Ordem de Venda'
        when '4' then 'Remessa'
        when '5' then 'Fatura'
        when '6' then 'XML'
        else          'Desconhecida'
      end         as EtapaText,

      _Arq
}
