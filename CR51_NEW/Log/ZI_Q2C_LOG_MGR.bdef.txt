//
// Read-only no RAP: sem create/update/delete expostos.
// Registros são criados exclusivamente via INSERT direto na

// BO independente — sem composition, sem lock dependent.
// -----------------------------------------------------------------------
// authorization não declarada — acesso controlado pelo @AccessControl.authorizationCheck: #NOT_REQUIRED no DDLS
// authorization master ( instance ) exigiria implementation class para handler de autorização — LOG não tem classe
define behavior for ZI_Q2C_LOG_MGR alias LogMgr
  persistent table ztbn_q2c_log_mgr
  lock master
{
  field ( readonly ) Pedido; Bandeira; Datum; Uzeit; Etapa; Mensagem; Ernam;

  mapping for ztbn_q2c_log_mgr
  {
    Pedido   = pedido;
    Bandeira = bandeira;
    Datum    = datum;
    Uzeit    = uzeit;
    Etapa    = etapa;
    Mensagem = mensagem;
    Ernam    = ernam;
  }
}
