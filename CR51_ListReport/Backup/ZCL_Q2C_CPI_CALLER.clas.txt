*&---------------------------------------------------------------------*
*& Class ZCL_Q2C_CPI_CALLER
*&---------------------------------------------------------------------*
*& Responsável por chamar o CPI para reprocessamento de arquivos
*& - Lê CabecArq (já em Base64) diretamente da tabela via SELECT SINGLE
*& - Monta payload JSON e envia via HTTP para o iFlow CPI
*& - Retorna status da resposta
*&
*& PAYLOAD enviado ao CPI (/http/mgr/reprocessamento):
*&   {
*&     "filename" : "<tipo_doc>",    <- campo TipoDoc do registro
*&     "conteudo" : "<base64>"       <- campo CabecArq (já em Base64)
*&   }
*&
*& Para adicionar/remover campos do payload: editar apenas o método
*& BUILD_PAYLOAD. A assinatura de call_cpi_reprocess não precisa mudar.
*&---------------------------------------------------------------------*

CLASS zcl_q2c_cpi_caller DEFINITION
  PUBLIC
  FINAL
  CREATE PUBLIC.

  PUBLIC SECTION.
    " Detalhe do último erro de comunicação CPI.
    " Disponível no CATCH cx_root do behavior pool via cpi_caller->mv_error_detail.
    DATA mv_error_detail TYPE string READ-ONLY.

    METHODS call_cpi_reprocess
      IMPORTING
        pedido   TYPE ztbq2c_arq_mgr-pedido
        bandeira TYPE ztbq2c_arq_mgr-bandeira
        tipo_doc TYPE ztbq2c_arq_mgr-tipo_doc
      EXPORTING
        ev_status TYPE string.

  PRIVATE SECTION.
    " Destino RFC configurado na SM59 (tipo G - HTTP para sistema externo)
    " Host, porta (443) e path (/http/mgr/reprocessamento) já definidos no destino.
    CONSTANTS c_destination TYPE rfcdest VALUE 'CPI_MGR_REPROCESSO'.

    " Estrutura interna com todos os campos necessários para o payload.
    " Para adicionar um campo: inclua-o aqui e use-o em build_payload.
    TYPES: BEGIN OF ty_payload_fields,
             tipo_doc  TYPE ztbq2c_arq_mgr-tipo_doc,
             cabec_arq TYPE ztbq2c_arq_mgr-cabec_arq,
           END OF ty_payload_fields.

    " Monta o JSON a ser enviado ao CPI.
    " Único método que precisa mudar quando o contrato do iFlow muda.
    METHODS build_payload
      IMPORTING
        is_fields         TYPE ty_payload_fields
      RETURNING
        VALUE(rv_payload) TYPE string.

    " Executa a chamada HTTP. Erros sinalizados via mv_error_detail (não lança exceção).
    " rv_response vazio + mv_error_detail preenchido = chamada com erro.
    METHODS execute_http_call
      IMPORTING
        payload            TYPE string
      RETURNING
        VALUE(rv_response) TYPE string.

ENDCLASS.

CLASS zcl_q2c_cpi_caller IMPLEMENTATION.

  METHOD call_cpi_reprocess.
    CLEAR me->mv_error_detail.

    " 1. Ler CabecArq da tabela — já está em Base64, não precisa converter
    SELECT SINGLE tipo_doc, cabec_arq
      FROM ztbq2c_arq_mgr
      WHERE pedido   = @pedido
        AND bandeira = @bandeira
      INTO @DATA(ls_arq).

    IF sy-subrc <> 0.
      me->mv_error_detail = |Registro não encontrado: Pedido={ pedido } Bandeira={ bandeira }|.
      RAISE EXCEPTION TYPE cx_sy_no_handler.
    ENDIF.

    " 2. Montar payload — alterar build_payload se o contrato do iFlow mudar
    DATA(lv_payload) = build_payload(
      VALUE #(
        tipo_doc  = ls_arq-tipo_doc
        cabec_arq = ls_arq-cabec_arq
      )
    ).

    " 3. Enviar ao CPI
    DATA(lv_response) = execute_http_call( lv_payload ).

    IF me->mv_error_detail IS NOT INITIAL.
      RAISE EXCEPTION TYPE cx_sy_no_handler.
    ENDIF.

    ev_status = lv_response.

  ENDMETHOD.

  METHOD build_payload.
    " ==========================================================
    " CONTRATO DO IFLOW CPI (/http/mgr/reprocessamento):
    "   "filename" → TipoDoc  (ex: PMDREN, PMDFORD)
    "   "conteudo" → CabecArq (Base64 — já gravado assim na tabela)
    "
    " Para adicionar campo:
    "   1. Adicione o campo em ty_payload_fields
    "   2. Passe-o no VALUE # em call_cpi_reprocess
    "   3. Inclua-o na string abaixo
    " ==========================================================
    rv_payload = |\{ "filename": "{ is_fields-tipo_doc }", | &
                 |"conteudo": "{ is_fields-cabec_arq }" \}|.
  ENDMETHOD.

  METHOD execute_http_call.
    DATA: lo_client    TYPE REF TO if_http_client,
          lv_body_xstr TYPE xstring,
          lv_http_code TYPE i,
          lv_reason    TYPE string.

    " 1. Criar cliente HTTP via RFC Destination (SM59: CPI_MGR_REPROCESSO, tipo G)
    cl_http_client=>create_by_destination(
      EXPORTING
        destination              = c_destination
      IMPORTING
        client                   = lo_client
      EXCEPTIONS
        argument_not_found       = 1
        destination_not_found    = 2
        destination_no_authority = 3
        plugin_not_active        = 4
        internal_error           = 5
        OTHERS                   = 6
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Destino RFC '{ c_destination }' não encontrado/acessível (subrc={ sy-subrc })|.
      RETURN.
    ENDIF.

    " 2. Método POST + Content-Type JSON
    lo_client->request->set_method( if_http_request=>co_request_method_post ).
    lo_client->request->set_header_field( name  = 'Content-Type'
                                          value = 'application/json' ).

    " 3. Converter payload JSON string → xstring e setar como body
    cl_bcs_convert=>string_to_xstring(
      EXPORTING iv_string  = payload
      RECEIVING ev_xstring = lv_body_xstr
    ).
    lo_client->request->set_data( lv_body_xstr ).

    " 4. Enviar
    lo_client->send(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        OTHERS                     = 3
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Falha de comunicação HTTP ao enviar para CPI (subrc={ sy-subrc })|.
      lo_client->close( ).
      RETURN.
    ENDIF.

    " 5. Receber resposta
    lo_client->receive(
      EXCEPTIONS
        http_communication_failure = 1
        http_invalid_state         = 2
        http_processing_failed     = 3
        OTHERS                     = 4
    ).
    IF sy-subrc <> 0.
      me->mv_error_detail = |Falha ao receber resposta do CPI (subrc={ sy-subrc })|.
      lo_client->close( ).
      RETURN.
    ENDIF.

    " 6. Verificar HTTP status code
    lo_client->response->get_status(
      IMPORTING
        code   = lv_http_code
        reason = lv_reason
    ).

    rv_response = lo_client->response->get_cdata( ).
    lo_client->close( ).

    " HTTP != 2xx = erro de negócio CPI
    IF lv_http_code < 200 OR lv_http_code >= 300.
      me->mv_error_detail = |CPI HTTP { lv_http_code } ({ lv_reason }): { rv_response }|.
      CLEAR rv_response.
    ENDIF.

  ENDMETHOD.

ENDCLASS.

ENDCLASS.
