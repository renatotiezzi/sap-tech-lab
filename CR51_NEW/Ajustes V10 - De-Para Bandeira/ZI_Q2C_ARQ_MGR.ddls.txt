@AbapCatalog.viewEnhancementCategory: [#NONE]
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Interface ARQ MGR - Reprocessamento'
@Metadata.ignorePropagatedAnnotations: true
@ObjectModel.usageType: {
  serviceQuality: #X,
  sizeCategory: #S,
  dataClass: #MIXED
}
define root view entity ZI_Q2C_ARQ_MGR
  as select from ztbq2c_arq_mgr as arq
{
  key arq.pedido      as Pedido,
  key arq.bandeira    as Bandeira,
      // De-Para tipoArquivo (campo tecnico) -> operacao (descricao exibida)
      cast(
        case arq.bandeira
          when 'PMDDSOP'     then 'PFCO_Volvo'
          when 'PMDDSOX'     then 'PRCO_Volvo'
          when 'PMDDAFDSOP'  then 'PFCO_Daf'
          when 'PMDDAFDSOX'  then 'PRCO_Daf'
          when 'PMDVOLV'     then 'DSH_Volvo'
          when 'PMDREN'      then 'DSH_Renault'
          when 'PMDFORD'     then 'DSH_Ford'
          when 'PMDDPASC'    then 'DSH_DPaschoal'
          when 'PMDJCB'      then 'DSH_JCB'
          when 'PMDDASA'     then 'DSH_DASA'
          when 'PMDHYDRO'    then 'DSH_HYDRO'
          else arq.bandeira
        end as abap.char( 20 )
      )               as BandeiraDesc,
      arq.tipo_doc    as TipoDoc,
      arq.arquivo     as Arquivo,
      arq.conteudo    as Conteudo,
      arq.status      as Status,
      case arq.status
        when 'ERRO'             then 1
        when 'PROCESSADO'       then 3
        when 'CANCELADO'        then 0
        when 'CRIADO'           then 5
        else 0
      end             as StatusCriticality,
      arq.tentativas  as Tentativas,
      arq.datum       as Datum,
      arq.uzeit       as Uzeit,
      arq.ernam       as Ernam,
      arq.ultimo_erro as UltimoErro
}
